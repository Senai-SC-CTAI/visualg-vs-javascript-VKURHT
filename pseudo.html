inicio

leia(valor)
escreval("Digite a sua opção de assinante!")
escreval("1) Assinante Premium, ganha 20% de desconto e frete grátis;")
escreval("2) Assinante Gold, ganha 20% de desconto mas paga frete;")
escreval("3) Assinante Silver, ganha 10% de desconto mas paga frete;")

se (assinante = "1") entao
desconto <- (valor * 0.20)

fimse

se (assinante = "2") entao
desconto <- (valor * 0.20)
pagar <- (valor - desconto) + 12
escreva("O valor do produto de assinante gold com todos os benefícios ficará R$",pagar)
fimse

se (assinante = "3") entao
desconto <- (valor * 0.10)
pagar <- (valor - desconto) + 12
escreva("O valor do produto de assinante silver com todos os benefícios ficará R$",pagar)
fimse
